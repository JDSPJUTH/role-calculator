  <div class="role" id="upper">

    <h3>Upper Class</h3>

    <div class="controls">

      <button onclick="updateRole('upper', -1)">–</button>

      <span id="upper-value">50</span>

      <button onclick="updateRole('upper', +1)">+</button>

    </div>

  </div>


  <div class="role" id="middle">

    <h3>Middle Class</h3>

    <div class="controls">

      <button onclick="updateRole('middle', -1)">–</button>

      <span id="middle-value">30</span>

      <button onclick="updateRole('middle', +1)">+</button>

    </div>

  </div>


  <div class="role" id="working">

    <h3>Working Class</h3>

    <div class="controls">

      <button onclick="updateRole('working', -1)">–</button>

      <span id="working-value">20</span>

      <button onclick="updateRole('working', +1)">+</button>

    </div>

  </div>


  <div class="role" id="lower">

    <h3>Lower Class</h3>

    <div class="controls">

      <button onclick="updateRole('lower', -1)">–</button>

      <span id="lower-value">10</span>

      <button onclick="updateRole('lower', +1)">+</button>

    </div>

  </div>


  <div class="global-controls">

    <button onclick="updateAll(-1)">-1 All</button>

    <button onclick="updateAll(+1)">+1 All</button>

  </div>


  <script>

    // Initialize role values

    const roles = {

      upper: 50,

      middle: 30,

      working: 20,

      lower: 10

    };


    // Update the displayed values for all roles

    function updateDisplay() {

      for (let role in roles) {

        document.getElementById(role + '-value').textContent = roles[role];

      }

    }


    // Update a single role by a delta (+1 or -1)

    function updateRole(role, delta) {

      roles[role] += delta;

      updateDisplay();

    }


    // Update all roles by a delta (+1 or -1)

    function updateAll(delta) {

      for (let role in roles) {

        roles[role] += delta;

      }

      updateDisplay();

    }

  </script>
